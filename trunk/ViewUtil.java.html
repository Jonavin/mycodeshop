    public static final Map<String, Object> extractFormValueByTag(ViewGroup form, String ... viewTags) {
        Map<String, Object> map = new HashMap<String, Object>();
        for(String tag : viewTags) {
            View view = form.findViewWithTag(tag);
            if(view instanceof TextView) {
                String value = ((TextView) view).getText().toString();
                map.put(tag, value);
            } else if(view instanceof Spinner) {
                Spinner spn = (Spinner)view;
                int pos = spn.getSelectedItemPosition();
                map.put(tag, pos);
            }
        }
        return map;
    }

    }




    public static final PopupWindow createPopupWindow(Context context, String title, View mainView, float widthPercent,
            float heightPercent, Button... buttons) {
        mainView.requestFocus();
        Activity activity = (Activity) context;
        // 设置弹出窗口长宽
        DisplayMetrics dm = new DisplayMetrics();
        activity.getWindowManager().getDefaultDisplay().getMetrics(dm);
        int width = (int) (dm.widthPixels * widthPercent);
        int height = (int) (dm.heightPixels * heightPercent);
        int btnWidth = width / 3;
        // 获取弹出窗口布局
        View contentView = activity.getLayoutInflater().inflate(R.layout.popup_window, null, false);
        // 设置弹出窗口标题
        LinearLayout layPopwinHead = (LinearLayout) contentView.findViewById(R.id.layPopwinHead);
        TextView txtTitle = (TextView) contentView.findViewById(R.id.txtPopwinTitle);
        if (title != null && !"".equals(title)) {
            txtTitle.setText(title);
        } else {
            ((LinearLayout.LayoutParams) layPopwinHead.getLayoutParams()).height = 15;
            txtTitle.setVisibility(View.GONE);
        }
        // 窗口主要内容区域
        LinearLayout layPopwinMain = (LinearLayout) contentView.findViewById(R.id.layPopwinMain);
        LinearLayout.LayoutParams prm4MainView = new LinearLayout.LayoutParams(LinearLayout.LayoutParams.FILL_PARENT, LinearLayout.LayoutParams.FILL_PARENT);
        mainView.setLayoutParams(prm4MainView);
        layPopwinMain.addView(mainView);
        // 窗口按钮区域
        LinearLayout layPopwinBottom = (LinearLayout) contentView.findViewById(R.id.layPopwinBottom);
        LinearLayout.LayoutParams btnPrm = new LinearLayout.LayoutParams(LinearLayout.LayoutParams.FILL_PARENT,
                LinearLayout.LayoutParams.WRAP_CONTENT);
        btnPrm.weight= 1;
        int i = 0;
        for (Button btn : buttons) {
            LinearLayout btnLL = new LinearLayout(context);
            LinearLayout.LayoutParams btnPrm1 = new LinearLayout.LayoutParams(LinearLayout.LayoutParams.WRAP_CONTENT,
                    LinearLayout.LayoutParams.WRAP_CONTENT);
            btnPrm1.weight= 1;
            if(i!=buttons.length-1&&buttons.length!=1)
                btnLL.setPadding(0, 0, 1, 0);
            btnLL.addView(btn, btnPrm1);
            layPopwinBottom.addView(btnLL, btnPrm);
            btn.setTextColor(Color.WHITE);
            btn.setBackgroundResource(R.drawable.btn_popwindow);
            i++;
        }

        PopupWindow win = new PopupWindow(contentView,width, height, true);

        win.setAnimationStyle(R.style.popwindow_anim);
//        win.setBackgroundDrawable(context.getResources().getDrawable(R.drawable.pop_win_bg));
        win.update();

        return win;
    }

    /* 判断是否是内网IP */
    public static final boolean isInnerIP(String ipAddress) {
        boolean isInnerIp = false;
        long ipNum = getIpNum(ipAddress);
        /**
         * 私有IP：A类 10.0.0.0-10.255.255.255 B类 172.16.0.0-172.31.255.255 C类
         * 192.168.0.0-192.168.255.255 当然，还有127这个网段是环回地址
         */
        long aBegin = getIpNum("10.0.0.0");
        long aEnd = getIpNum("10.255.255.255");
        long bBegin = getIpNum("172.16.0.0");
        long bEnd = getIpNum("172.31.255.255");
        long cBegin = getIpNum("192.168.0.0");
        long cEnd = getIpNum("192.168.255.255");
        isInnerIp = isInner(ipNum, aBegin, aEnd) || isInner(ipNum, bBegin, bEnd)
                || isInner(ipNum, cBegin, cEnd) || ipAddress.equals("127.0.0.1");
        return isInnerIp;
    }

    /* 获取IP数 */
    private static final long getIpNum(String ipAddress) {
        String[] ip = ipAddress.split("\\.");
        long a = Integer.parseInt(ip[0]);
        long b = Integer.parseInt(ip[1]);
        long c = Integer.parseInt(ip[2]);
        long d = 0;
        try {
            d = Integer.parseInt(ip[3]);
        } catch (Exception ex) {

        }

        long ipNum = a * 256 * 256 * 256 + b * 256 * 256 + c * 256 + d;
        return ipNum;
    }

    private static final boolean isInner(long userIp, long begin, long end) {
        return (userIp >= begin) && (userIp <= end);
    }

    /**
     * 用配置的样式渲染Value
     * @param view
     * @param fieldVO
     * @author rongxinhua
     */
    public static void stylingValue(View view, MmPageAreaFieldVO fieldVO) {
        if(fieldVO == null) {
            return;
        }
        //可见性
        if(fieldVO.getIsDisplay() == null || !fieldVO.getIsDisplay().equals("1")) {
            view.setVisibility(View.GONE);
        }
        if(view instanceof TextView) {
            TextView tv = (TextView)view;
            //字体颜色
            if(fieldVO.getFontColor() != null && !fieldVO.getFontColor().equals("")) {
                tv.setTextColor(Color.parseColor(fieldVO.getFontColor()));
            }
            //字体大小
            if(fieldVO.getFontSize() != null && !fieldVO.getFontSize().equals("")) {
                tv.setTextSize(Float.parseFloat(fieldVO.getFontSize()));
            }
            //字体加粗
            if(fieldVO.getIsBold() != null && fieldVO.getIsBold().equals("1")) {
                drawBold(view);
            }
        }
    }

    /**
     * 线性布局参数
     * @param prms 依次：width, height, weight, leftMargin, topMargin, rightMargin, bottomMargin
     * @return
     */
    public static LinearLayout.LayoutParams newLinePrm(int ... prms) {
        if(prms.length < 2) {
            throw new RuntimeException("只少两个参数！！");
        }
        LinearLayout.LayoutParams prm = new LinearLayout.LayoutParams(prms[0], prms[1]);
        try {
            prm.weight = prms[2];
            prm.leftMargin = prms[3];
            prm.topMargin = prms[4];
            prm.rightMargin = prms[5];
            prm.bottomMargin = prms[6];
        } catch(Exception e) {}
        return prm;
    }

    /**
     * 创建加载进度条
     * @param context
     * @param visibility 默认可见性， View.GONE, View.VISIABLE, View.INVISIABLE
     * @return
     * @author rongxinhua
     */
    public static LinearLayout createLoadingBar(Context context, String text, int visibility) {
        LinearLayout loadingLayout;
        LinearLayout layout = new LinearLayout(context);  
        // 设置布局 水平方向  
        layout.setOrientation(LinearLayout.HORIZONTAL);  
        // 进度条  
        ProgressBar progressBar = new ProgressBar(context);
        progressBar.setIndeterminateDrawable(context.getResources().getDrawable(R.drawable.comp_progressbar));
        // 进度条显示位置  
        progressBar.setPadding(0, 0, 15, 0);  
        // 把进度条加入到layout中  
        LinearLayout.LayoutParams mLayoutParams = new LinearLayout.LayoutParams(LinearLayout.LayoutParams.WRAP_CONTENT, LinearLayout.LayoutParams.WRAP_CONTENT);            
        layout.addView(progressBar, mLayoutParams);  
        // 文本内容  
        TextView textView = new TextView(context);  
        textView.setText(text);
        textView.setTextColor(Color.GRAY);
        textView.setGravity(Gravity.CENTER_VERTICAL);  
        //把文本加入到layout中  
        LinearLayout.LayoutParams FFlayoutParams = new LinearLayout.LayoutParams(LinearLayout.LayoutParams.FILL_PARENT, LinearLayout.LayoutParams.WRAP_CONTENT);  
        layout.addView(textView, FFlayoutParams);  
        // 设置layout的重力方向，即对齐方式是  
        layout.setGravity(Gravity.CENTER);              
        loadingLayout = new LinearLayout(context);  
        loadingLayout.addView(layout, mLayoutParams);  
        loadingLayout.setGravity(Gravity.CENTER);
        loadingLayout.setVisibility(visibility);
        return loadingLayout;
    }

    /**
     * 传入一个视图或ViewGroup的子类 如果是一个TextView 则加粗
     * 如果是一个LinearLayout等布局，则把他下面的TextView都加粗
     * 
     * @param v
     */
    public static void drawBold(View v) {
        if (v instanceof ViewGroup) {
            int length = ((ViewGroup) v).getChildCount();
            for (int i = 0; i < length; i++) {
                View child = ((ViewGroup) v).getChildAt(i);
                if (child instanceof TextView) {
                    TextView tv = (TextView) child;
                    TextPaint tp = tv.getPaint();
                    tp.setFakeBoldText(true);
                } else if (child instanceof ViewGroup) {
                    drawBold(child);
                }
            }
        } else if (v instanceof TextView) {
            TextView tv = (TextView) v;
            TextPaint tp = tv.getPaint();
            tp.setFakeBoldText(true);
        }
    }
